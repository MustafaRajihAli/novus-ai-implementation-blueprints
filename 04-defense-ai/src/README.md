Threat Detection Engine

```json
# src/services/threat_detection.py"""Multi-source threat fusion engine. Ingests events from radar,seismic, visual, and cyber sources, cross-correlates within asliding time window, and classifies composite threat severity."""import asynciofrom datetime import datetime, timedeltafrom collections import defaultdictfrom src.models.threat import ThreatEvent, FusedThreat, ThreatSeverityfrom src.ml.trajectory_predictor import TrajectoryPredictorimport torchFUSION_WINDOW_SEC = 120  # Events within 2 minutes are candidates for fusionSPATIAL_RADIUS_KM = 2.0  # Events within 2km are candidates for fusionclass ThreatDetectionEngine:    def __init__(self):        self.predictor = TrajectoryPredictor("models/trajectory_lstm_v2.pt")        self.event_buffer: list[ThreatEvent] = []    async def ingest_event(self, event: ThreatEvent) -> FusedThreat | None:        """        Add an event to the buffer and check for correlated threats.        Returns a FusedThreat if correlation is found.        """        self.event_buffer.append(event)        self._prune_old_events()        correlated = self._find_correlated_events(event)        if correlated:            return await self._build_fused_threat(event, correlated)        return None    def _find_correlated_events(self, anchor: ThreatEvent) -> list[ThreatEvent]:        """Find other recent events near the same geographic location."""        cutoff = anchor.timestamp - timedelta(seconds=FUSION_WINDOW_SEC)        candidates = []        for e in self.event_buffer:            if e.event_id == anchor.event_id:                continue            if e.timestamp < cutoff:                continue            dist = haversine_km(anchor.location, e.location)            if dist <= SPATIAL_RADIUS_KM:                candidates.append(e)        return candidates    async def _build_fused_threat(        self, anchor: ThreatEvent, correlated: list[ThreatEvent]    ) -> FusedThreat:        """Combine correlated events and predict trajectory."""        all_events = [anchor] + correlated        sources = list({e.source for e in all_events})        avg_confidence = sum(e.confidence for e in all_events) / len(all_events)        # More sources = higher severity baseline        if len(sources) >= 3:            severity = ThreatSeverity.CRITICAL        elif len(sources) == 2:            severity = ThreatSeverity.ELEVATED        else:            severity = ThreatSeverity.MONITORED        trajectory = await self.predictor.predict(            events=all_events, horizon_minutes=120        )        return FusedThreat(            fused_sources=sources,            severity=severity,            avg_confidence=avg_confidence,            predicted_trajectory=trajectory,            requires_authorization=severity in (ThreatSeverity.ELEVATED, ThreatSeverity.CRITICAL),        )    def _prune_old_events(self):        cutoff = datetime.utcnow() - timedelta(seconds=FUSION_WINDOW_SEC * 2)        self.event_buffer = [e for e in self.event_buffer if e.timestamp > cutoff]def haversine_km(loc1: dict, loc2: dict) -> float:    """Calculate distance between two lat/lon points in kilometers."""    import math    R = 6371    lat1, lon1 = math.radians(loc1["lat"]), math.radians(loc1["lon"])    lat2, lon2 = math.radians(loc2["lat"]), math.radians(loc2["lon"])    dlat, dlon = lat2 - lat1, lon2 - lon1    a = math.sin(dlat/2)**2 + math.cos(lat1)*math.cos(lat2)*math.sin(dlon/2)**2    return R * 2 * math.asin(math.sqrt(a))
```

Trajectory Predictor

```json
# src/ml/trajectory_predictor.py"""Spatiotemporal LSTM that predicts threat actor movementtrajectory from a sequence of observed position events."""import torchimport torch.nn as nnimport numpy as npfrom src.models.threat import ThreatEventclass TrajectoryLSTM(nn.Module):    def __init__(self, input_dim=6, hidden_dim=128, output_dim=2, num_layers=2):        super().__init__()        # Input: [lat, lon, speed_est, heading_est, source_encoded, confidence]        self.lstm = nn.LSTM(input_dim, hidden_dim, num_layers, batch_first=True)        self.head = nn.Sequential(            nn.Linear(hidden_dim, 64),            nn.ReLU(),            nn.Linear(64, output_dim)  # Predict [delta_lat, delta_lon]        )    def forward(self, x):        out, _ = self.lstm(x)        return self.head(out[:, -1, :])class TrajectoryPredictor:    def __init__(self, model_path: str):        self.model = TrajectoryLSTM()        self.model.load_state_dict(torch.load(model_path, map_location="cpu"))        self.model.eval()        self.STEP_MINUTES = 15    async def predict(self, events: list[ThreatEvent], horizon_minutes: int):        features = self._extract_features(events)        tensor = torch.FloatTensor(features).unsqueeze(0)        predictions = []        current_pos = [events[-1].location["lat"], events[-1].location["lon"]]        with torch.no_grad():            steps = horizon_minutes // self.STEP_MINUTES            for step in range(steps):                delta = self.model(tensor).squeeze().tolist()                current_pos[0] += delta[0]                current_pos[1] += delta[1]                predictions.append({                    "lat": round(current_pos[0], 6),                    "lon": round(current_pos[1], 6),                    "eta_min": (step + 1) * self.STEP_MINUTES                })                # Append prediction as next input step                new_feat = self._encode_prediction(current_pos, delta)                tensor = torch.cat([tensor[:, 1:, :],                                     torch.FloatTensor(new_feat).unsqueeze(0).unsqueeze(0)], dim=1)        return predictions    def _extract_features(self, events: list[ThreatEvent]) -> list:        return [[            e.location["lat"], e.location["lon"],            e.speed_est or 0.0, e.heading_est or 0.0,            hash(e.source) % 10 / 10.0, e.confidence        ] for e in events]    def _encode_prediction(self, pos, delta):        speed = (delta[0]**2 + delta[1]**2)**0.5 * 111  # rough km/step        return [pos[0], pos[1], speed, 0.0, 0.5, 0.9]
```
