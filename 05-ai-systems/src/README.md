```json
# src/services/ai_orchestrator.py"""Central AI orchestration service. Routes tasks to the correctAI module (LLM, vision, anomaly detection) and handles retries,caching, and structured output validation."""import hashlibimport jsonfrom functools import lru_cachefrom langchain_openai import ChatOpenAIfrom langchain.agents import create_tool_calling_agent, AgentExecutorfrom langchain_core.prompts import ChatPromptTemplatefrom src.services.rag_retriever import RAGRetrieverfrom src.models.ai_task import AITask, AIResultfrom src.utils.redis_cache import semantic_cache_get, semantic_cache_setSUPPORTED_MODELS = {    "gpt-4o": ChatOpenAI(model="gpt-4o"),    "gpt-4o-mini": ChatOpenAI(model="gpt-4o-mini"),}class AIOrchestrator:    def __init__(self, model_name: str = "gpt-4o"):        self.llm = SUPPORTED_MODELS[model_name]        self.retriever = RAGRetriever()    async def execute_task(self, task: AITask) -> AIResult:        """        Main entry point. Checks semantic cache first, then        runs the appropriate AI pipeline.        """        cache_key = self._cache_key(task)        cached = await semantic_cache_get(cache_key)        if cached:            return AIResult(**cached, from_cache=True)        if task.knowledge_base_id:            context = await self.retriever.retrieve(                query=task.input, kb_id=task.knowledge_base_id, top_k=5            )        else:            context = []        result = await self._run_agent(task, context)        await semantic_cache_set(cache_key, result.dict(), ttl=3600)        return result    async def _run_agent(self, task: AITask, context: list) -> AIResult:        """Build and run LangChain agent with tools and structured output."""        system_prompt = self._build_system_prompt(task, context)        prompt = ChatPromptTemplate.from_messages([            ("system", system_prompt),            ("human", "{input}"),            ("placeholder", "{agent_scratchpad}"),        ])        tools = self._load_tools(task.tools or [])        if task.structured_output_schema:            # Force structured JSON output            llm_with_schema = self.llm.with_structured_output(                task.structured_output_schema, method="json_mode"            )            response = await llm_with_schema.ainvoke(                [("system", system_prompt), ("human", task.input)]            )            return AIResult(result=response, confidence=0.9, tool_calls=[])        agent = create_tool_calling_agent(self.llm, tools, prompt)        executor = AgentExecutor(agent=agent, tools=tools, max_iterations=5)        response = await executor.ainvoke({"input": task.input})        return AIResult(            result=response["output"],            confidence=self._estimate_confidence(response),            tool_calls=response.get("intermediate_steps", [])        )    def _build_system_prompt(self, task: AITask, context: list) -> str:        base = "You are a professional AI assistant. Be precise and factual."        if context:            ctx_text = "\n".join([f"- {c.page_content}" for c in context])            base += f"\n\nRelevant context:\n{ctx_text}"        return base    def _cache_key(self, task: AITask) -> str:        h = hashlib.sha256(            json.dumps({"input": task.input, "type": task.task_type}, sort_keys=True)            .encode()        ).hexdigest()        return f"ai_task:{h}"    def _load_tools(self, tool_names: list):        from src.tools import TOOL_REGISTRY        return [TOOL_REGISTRY[name] for name in tool_names if name in TOOL_REGISTRY]    def _estimate_confidence(self, response: dict) -> float:        # Heuristic: fewer tool retries = higher confidence        steps = len(response.get("intermediate_steps", []))        return max(0.6, 1.0 - (steps * 0.1))
```

Rag Retriever

```dockerfile
# src/services/rag_retriever.py"""RAG retriever using pgvector for semantic search overingested knowledge base documents."""import asyncpgfrom sentence_transformers import SentenceTransformerfrom src.models.knowledge import DocumentChunkEMBEDDING_MODEL = "sentence-transformers/all-MiniLM-L6-v2"TOP_K_DEFAULT = 5class RAGRetriever:    def __init__(self):        self.embedder = SentenceTransformer(EMBEDDING_MODEL)    async def retrieve(        self, query: str, kb_id: str, top_k: int = TOP_K_DEFAULT    ) -> list[DocumentChunk]:        """Embed the query and find the most semantically similar chunks."""        query_embedding = self.embedder.encode(query).tolist()        async with asyncpg.connect(DB_URL) as conn:            rows = await conn.fetch(                """                SELECT id, content, metadata,                       1 - (embedding <=> $1::vector) AS similarity                FROM document_chunks                WHERE kb_id = $2                ORDER BY embedding <=> $1::vector                LIMIT $3                """,                query_embedding, kb_id, top_k            )        return [            DocumentChunk(                chunk_id=str(row["id"]),                page_content=row["content"],                metadata=row["metadata"],                similarity=row["similarity"]            )            for row in rows if row["similarity"] > 0.65        ]    async def ingest_document(self, kb_id: str, document: dict) -> int:        """        Chunk a document, embed each chunk, and store in pgvector.        Returns number of chunks created.        """        chunks = self._chunk_text(document["content"])        embeddings = self.embedder.encode(chunks).tolist()        async with asyncpg.connect(DB_URL) as conn:            for chunk, emb in zip(chunks, embeddings):                await conn.execute(                    """                    INSERT INTO document_chunks (kb_id, content, embedding, metadata)                    VALUES ($1, $2, $3::vector, $4)                    """,                    kb_id, chunk, emb, document.get("metadata", {})                )        return len(chunks)    def _chunk_text(self, text: str, chunk_size: int = 400, overlap: int = 80) -> list[str]:        """Sliding window chunking with overlap for context continuity."""        words = text.split()        chunks = []        for i in range(0, len(words), chunk_size - overlap):            chunk = " ".join(words[i:i + chunk_size])            if chunk:                chunks.append(chunk)        return chunks
```
